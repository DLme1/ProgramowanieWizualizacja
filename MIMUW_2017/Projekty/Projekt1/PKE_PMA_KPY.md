wymiana <- function(co_sprzedaj,
                    co_kup,
                    ile_sprzedaj,
                    ile_kup) {
                    zagroda[co_kup] <<- zagroda[co_kup] + ile_kup
                    zagroda[co_sprzedaj] <<- zagroda[co_sprzedaj] - ile_sprzedaj
                    stado[co_kup] <<- stado[co_kup] - ile_kup
                    stado[co_sprzedaj] <<- stado[co_sprzedaj] + ile_sprzedaj
                    return(zagroda)
                    }
