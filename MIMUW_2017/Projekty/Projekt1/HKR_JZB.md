dodaj_lub_odejmij_zwierzeta <- function(co, jak='+',stado,stado_glowne){
  if (jak=='+'){ # dodajemy do stada
    stado <- stado + co
    stado_glowne <- stado_glowne - co
  } else { # odejmujemy ze stada
