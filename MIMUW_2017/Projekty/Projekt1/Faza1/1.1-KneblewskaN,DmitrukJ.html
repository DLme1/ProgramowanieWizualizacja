#v-wektor zwierząt który placi za zakup
#funkcje stosujemy tak, aby nie przekroczyć limitu zwierząt
zaplac <- function(a,v){
  x<-z[a]
  for (i in c(5:1)) {
    if (w[i] >= x) {
      if (x/z[i]>=1){
        v[i] <- v[i]-x/z[i]
        v[a] <- v[a]+1
      }else { 
        v[i]<-v[i]-1
        v[a]<-v[a]+(z[i]/x)
      }
      break()
    }else{
      v[i] <- v[i]-(w[i]/z[i])
      x <- x-w[i]
    }
  }
  
  return(v)
}
